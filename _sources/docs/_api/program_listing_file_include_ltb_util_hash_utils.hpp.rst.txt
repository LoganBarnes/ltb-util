   template <class T>
   auto hash_combine(std::size_t const& seed, const T& v) -> std::size_t {
       std::hash<T> hasher;
       return seed ^ (hasher(v) + 0x9e3779b9u + (seed << 6u) + (seed >> 2u));
   }
