#include <chrono>

namespace ltb::util {

class Timer {
public:
    explicit Timer(std::string name = "", std::ostream* os = nullptr);
    auto start() -> void;
    auto millis_since_start() -> double;

private:
    std::string                                        name_;
    std::ostream*                                      ostream_;

};

public:
    explicit ScopedTimer(std::string name, std::ostream& os = std::cout);
    ~ScopedTimer();

private:
    Timer timer_;
};
